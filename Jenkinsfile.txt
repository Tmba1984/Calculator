pipeline{
	agent any
	
	stages{
		stage('Git-Checkout'){
			steps{
				echo "Checking out from Git Repo"
				git 'https://github.com/Tmba1984/Calculator.git'
			}
		
		}
		stage('Build'){
			steps{
				echo "Building the checking out project"
				bat 'mvn clean compile'
			}
		}
		stage('Test'){
			steps{
				echo "Running tests"
				bat 'mvn test'
			}
			
			post{
				always{
					junit '**/target/surefire-reports/TEST-*.xml'
				}
			}
		}

	}
}
